CREATE TABLE equipment (
	equipment_id SERIAL PRIMARY KEY,
	name VARCHAR(255) NOT NULL,
	type VARCHAR(255) NOT NULL,
	purchase_date DATE NOT NULL,
	last_maintenance_date DATE,
	maintenance_frequency VARCHAR(255),
	status VARCHAR(255)
)
